const FIXED_EXPENSE = {
  "Chi phí sinh hoạt": ["Ăn uống", "Coffee", "Mua sắm gia đình", "Online shopping", "Dating", "Pet", "Thể thao", "Di chuyển", "Game"],
  "Chi phí cố định": ["Tiền thuê nhà", "Điện nước", "Internet", "Bảo hiểm", "Học phí"],
  "Chi phí phát sinh": ["Nhậu", "Đám tiệc", "Quà cáp", "Y tế", "Du lịch", "Sửa xe", "Household", "Công việc", "Tín dụng", "Cho mượn"],
  "Đầu tư - Tiết kiệm": ["Quỹ tiết kiệm", "Chứng khoán / Vàng", "Quỹ khác"]
};
const FIXED_INCOME = {
     "Thu nhập": ["Lương chính", "Job phụ", "Job phụ 2"],
     "Lãi/Lời": ["Lãi tiết kiệm", "Lãi đầu tư", "Cổ tức"],
     "Thưởng/Quà": ["Thưởng", "Được tặng", "Trúng thưởng"],
     "Khác": ["Thu hồi nợ", "Bán đồ cũ"]
   };

export default {
  async fetch(request, env, ctx) {
    const url = new URL(request.url);

    // ==========================================
    // 1. NHÁNH API CHO TELEGRAM MINI APP
    // ==========================================
    if (url.pathname.startsWith('/api/')) {
      const headers = { 
        'Content-Type': 'application/json', 
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'GET, PATCH, OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type, Authorization'
      };

      if (request.method === 'OPTIONS') return new Response(null, { headers });

      if (url.pathname === '/api/wallets' && request.method === 'GET') {
        const chatId = url.searchParams.get('chat_id');
        if (!chatId) return new Response(JSON.stringify({ error: "Missing chat_id" }), { status: 400, headers });
        const accounts = await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`);
        return new Response(JSON.stringify(accounts || []), { status: 200, headers });
      }

      if (url.pathname === '/api/categories' && request.method === 'GET') {
        const chatId = url.searchParams.get('chat_id');
        if (!chatId) return new Response(JSON.stringify({ error: "Missing chat_id" }), { status: 400, headers });
        let categories = await supabaseFetch(env, `categories?chat_id=eq.${chatId}`);
        
        // Tự động khởi tạo danh mục mặc định nếu chưa có (ví dụ khi mới reset DB)
        if (!categories || categories.length === 0) {
          const defaultCategories = [];
          for (const [level1, level2Array] of Object.entries(FIXED_EXPENSE)) {
            level2Array.forEach(level2 => {
              defaultCategories.push({ chat_id: chatId, transaction_type: 'chi', level_1: level1, level_2: level2 });
            });
          }
          for (const [level1, level2Array] of Object.entries(FIXED_INCOME)) {
            level2Array.forEach(level2 => {
              defaultCategories.push({ chat_id: chatId, transaction_type: 'thu', level_1: level1, level_2: level2 });
            });
          }
          await supabaseFetch(env, 'categories', 'POST', defaultCategories);
          categories = await supabaseFetch(env, `categories?chat_id=eq.${chatId}`);
        }
        return new Response(JSON.stringify(categories || []), { status: 200, headers });
      }

      if (url.pathname === '/api/categories' && request.method === 'POST') {
        try {
          const body = await request.json();
          const { chat_id, transaction_type, level_1, level_2 } = body;
          
          if (!chat_id || !transaction_type || !level_1 || !level_2) {
            return new Response(JSON.stringify({ error: "Missing required fields" }), { status: 400, headers });
          }

          // Kiểm tra trùng
          const existing = await supabaseFetch(env, `categories?chat_id=eq.${chat_id}&transaction_type=eq.${transaction_type}&level_1=eq.${encodeURIComponent(level_1)}&level_2=eq.${encodeURIComponent(level_2)}`);
          if (existing && existing.length > 0) {
            return new Response(JSON.stringify(existing[0]), { status: 200, headers });
          }

          const inserted = await supabaseFetch(env, 'categories', 'POST', {
            chat_id,
            transaction_type,
            level_1,
            level_2
          }, 'return=representation');

          const record = (Array.isArray(inserted) && inserted.length > 0) ? inserted[0] : (inserted || { chat_id, transaction_type, level_1, level_2 });
          return new Response(JSON.stringify(record), { status: 201, headers });
        } catch (err) {
          return new Response(JSON.stringify({ error: err.message }), { status: 500, headers });
        }
      }

      if (url.pathname === '/api/categories' && request.method === 'DELETE') {
        try {
          const body = await request.json();
          const { chat_id, transaction_type, level_1, level_2 } = body;
          
          if (!chat_id || !transaction_type || !level_1 || !level_2) {
            return new Response(JSON.stringify({ error: "Missing required fields" }), { status: 400, headers });
          }

          await supabaseFetch(env, `categories?chat_id=eq.${chat_id}&transaction_type=eq.${transaction_type}&level_1=eq.${encodeURIComponent(level_1)}&level_2=eq.${encodeURIComponent(level_2)}`, 'DELETE');
          return new Response(JSON.stringify({ success: true }), { status: 200, headers });
        } catch (err) {
          return new Response(JSON.stringify({ error: err.message }), { status: 500, headers });
        }
      }

      if (url.pathname === '/api/wallets' && request.method === 'PATCH') {
        try {
          const body = await request.json();
          if (!body.id) {
            return new Response(JSON.stringify({ error: "Missing account ID" }), { status: 400, headers });
          }

          // Cập nhật ví mặc định (is_default)
          if (body.is_default === true) {
            // 1. Lấy ví hiện tại để biết chat_id
            const currentAcc = await supabaseFetch(env, `accounts?id=eq.${body.id}`);
            if (currentAcc && currentAcc.length > 0) {
              const chatId = currentAcc[0].chat_id;
              // 2. Set tất cả ví của user này thành không mặc định
              await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`, 'PATCH', { is_default: false });
            }
          }

          // Chuẩn bị các trường cần cập nhật
          const updateFields = {};
          if (body.balance !== undefined) updateFields.balance = parseInt(body.balance);
          if (body.account_name !== undefined) updateFields.account_name = body.account_name.trim();
          if (body.is_default !== undefined) updateFields.is_default = body.is_default;

          const updateRes = await supabaseFetch(env, `accounts?id=eq.${body.id}`, 'PATCH', updateFields);
          if (updateRes !== null) {
            return new Response(JSON.stringify({ success: true }), { status: 200, headers });
          } else {
            return new Response(JSON.stringify({ error: "Database error" }), { status: 500, headers });
          }
        } catch (e) {
          return new Response(JSON.stringify({ error: e.message }), { status: 500, headers });
        }
      }

      if (url.pathname === '/api/dashboard' && request.method === 'GET') {
        const chatId = url.searchParams.get('chat_id');
        if (!chatId) return new Response(JSON.stringify({ error: "Missing chat_id" }), { status: 400, headers });

        try {
          const accounts = await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`);
          let totalBalance = 0;
          accounts?.forEach(acc => totalBalance += parseInt(acc.balance || 0));

          const now = new Date();
          const currentMonth = ("0" + (now.getMonth() + 1)).slice(-2) + "/" + now.getFullYear();
          
          // Lấy toàn bộ giao dịch của user (sắp xếp mới nhất lên đầu)
          const txs = await supabaseFetch(env, `transactions?chat_id=eq.${chatId}&order=created_at.desc`);

          let monthlyIncome = 0;
          let monthlyExpense = 0;
          let expenseByCategory = {};

          txs?.forEach(tx => {
            const amount = parseInt(tx.amount || 0);
            // Chỉ tính toán thu/chi của tháng hiện tại cho phần Dashboard
            const isCurrentMonth = tx.date && tx.date.endsWith(currentMonth);
            if (isCurrentMonth) {
              if (tx.type === 'thu') {
                monthlyIncome += amount;
              } else if (tx.type === 'chi') {
                monthlyExpense += amount;
                if (!expenseByCategory[tx.level1]) expenseByCategory[tx.level1] = 0;
                expenseByCategory[tx.level1] += amount;
              }
            }
          });

          const chartData = Object.keys(expenseByCategory).map(key => ({
            label: key,
            value: expenseByCategory[key]
          }));

          const dashboardData = {
            total_balance: totalBalance,
            monthly_income: monthlyIncome,
            monthly_expense: monthlyExpense,
            chart_data: chartData,
            recent_transactions: txs
          };

          return new Response(JSON.stringify(dashboardData), { status: 200, headers });
        } catch (e) {
          return new Response(JSON.stringify({ error: e.message }), { status: 500, headers });
        }
      }
    }

    // ==========================================
    // 2. NHÁNH WEBHOOK CHO TELEGRAM BOT
    // ==========================================
    if (request.method === 'POST') {
      try {
        const update = await request.json();
        // TRUYỀN ĐỦ 3 BIẾN ĐỂ CHẠY NGẦM KHÔNG LỖI
        if (update.message) {
          ctx.waitUntil(handleMessage(update.message, env, ctx));
        } else if (update.callback_query) {
          ctx.waitUntil(handleCallbackQuery(update.callback_query, env, ctx));
        }
        return new Response('OK', { status: 200 });
      } catch (error) {
        console.error('Error processing webhook:', error);
        return new Response('Error', { status: 500 });
      }
    }

    return new Response('Dancin CFO API is running!', { status: 200 });
  }
};

async function supabaseFetch(env, endpoint, method = 'GET', body = null, customPrefer = null) {
  const dbUrlRaw = env.SUPABASE_URL || env.supabase_url;
  const dbKey = env.SUPABASE_KEY || env.supabase_key;
  if (!dbUrlRaw || !dbKey) return null;

  const dbUrl = dbUrlRaw.replace(/\/$/, '');
  const url = `${dbUrl}/rest/v1/${endpoint}`;
  
  const options = {
    method: method,
    headers: {
      'apikey': dbKey,
      'Authorization': `Bearer ${dbKey}`,
      'Content-Type': 'application/json',
      // Nếu có truyền customPrefer thì dùng, nếu không thì tự nhận diện theo method
      'Prefer': customPrefer || (method === 'POST' ? 'return=minimal, resolution=ignore-duplicates' : 'return=representation')
    }
  };
  
  if (body && method !== 'GET') options.body = JSON.stringify(body);

  try {
    const response = await fetch(url, options);
    if (!response.ok) return null;
    // Bắt lỗi rỗng nếu API trả về HTTP 204 No Content (khi DELETE)
    const text = await response.text();
    return text ? JSON.parse(text) : null;
  } catch (e) {
    return null;
  }
}

async function sendMessage(env, chatId, text, keyboard = null) {
  const payload = { chat_id: chatId, text: text, parse_mode: 'HTML' };
  if (keyboard) payload.reply_markup = keyboard;
  
  await fetch(`https://api.telegram.org/bot${env.TelegramToken}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
}

function mapLevel1(input, type) {
  const clean = input.toLowerCase().trim();
  if (type === 'chi') {
    if (clean.includes('sinh hoat') || clean.includes('sinh hoạt')) return 'Chi phí sinh hoạt';
    if (clean.includes('co dinh') || clean.includes('cố định')) return 'Chi phí cố định';
    if (clean.includes('phat sinh') || clean.includes('phát sinh')) return 'Chi phí phát sinh';
    if (clean.includes('dau tu') || clean.includes('đầu tư') || clean.includes('tiet kiem') || clean.includes('tiết kiệm')) return 'Đầu tư - Tiết kiệm';
  } else {
    if (clean.includes('thu nhap') || clean.includes('thu nhập')) return 'Thu nhập';
    if (clean.includes('lai') || clean.includes('lãi') || clean.includes('loi') || clean.includes('lời')) return 'Lãi/Lời';
    if (clean.includes('thuong') || clean.includes('thưởng') || clean.includes('qua') || clean.includes('quà')) return 'Thưởng/Quà';
    if (clean.includes('khac') || clean.includes('khác')) return 'Khác';
  }
  return null;
}

async function getFilePath(env, fileId) {
  const url = `https://api.telegram.org/bot${env.TelegramToken}/getFile?file_id=${fileId}`;
  const response = await fetch(url);
  const data = await response.json();
  if (data.ok) {
    return data.result.file_path;
  }
  return null;
}

// LẤY TỆP TỪ TELEGRAM VÀ TRANS AUDIO //
async function transcribeAudio(env, filePath) {
  const fileUrl = `https://api.telegram.org/file/bot${env.TelegramToken}/${filePath}`;
  
  const audioResponse = await fetch(fileUrl);
  const audioBlob = await audioResponse.blob();

  const formData = new FormData();
  formData.append('file', audioBlob, 'audio.ogg');
  formData.append('model', 'whisper-large-v3');
  formData.append('prompt', 'Techcombank, thẻ tín dụng, tiền mặt, ăn sáng, cafe, trà sữa, nhậu, mua sắm, 50k, 100k, lương, thu nhập, chi phí sinh hoạt, hũ thiết yếu.');

  const groqKey = env.GROQ_API_KEY || env.GroqAPIkey || env.groq_api_key;
  
  const response = await fetch('https://api.groq.com/openai/v1/audio/transcriptions', {
    method: 'POST',
    headers: { 'Authorization': `Bearer ${groqKey}` },
    body: formData
  });

  const result = await response.json();
  return result.text || "";
}

async function handleMessage(message, env, ctx) {
  const chatId = message.chat.id.toString();
  const text = message.text || "";
  // Khai báo biến name ngay từ đầu để sử dụng cho mọi ngữ cảnh
  const name = message.from?.first_name || message.chat?.first_name || 'bạn';

  // 1. LỆNH /START (Đồng bộ & Bắt lỗi nghiêm ngặt)
  if (text === '/start') {
    await sendMessage(env, chatId, `⏳ Đang kết nối và thiết lập không gian cá nhân của bạn...`);

    // Tạo User
    await supabaseFetch(env, 'users', 'POST', { chat_id: chatId, name: name });

    // Kiểm tra xem đã có đủ 3 ví chưa, nếu chưa thì tạo mới
    let accs = await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`);
    if (!accs || accs.length < 3) {
      const defaultAccounts = [
        { chat_id: chatId, account_name: 'Tiền mặt', account_type: 'tien_mat', balance: 0, is_default: true },
        { chat_id: chatId, account_name: 'Techcombank', account_type: 'ngan_hang', balance: 0, is_default: false },
        { chat_id: chatId, account_name: 'Thẻ tín dụng', account_type: 'tin_dung', balance: 0, is_default: false }
      ];
      await supabaseFetch(env, 'accounts', 'POST', defaultAccounts);
      accs = await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`);
    }

    // 🌟 THÊM MỚI TẠI ĐÂY: KHỞI TẠO TỪ ĐIỂN DANH MỤC CÁ NHÂN
    let userCategories = await supabaseFetch(env, `categories?chat_id=eq.${chatId}`);
    
    // Nếu user chưa có danh mục nào (user mới), tiến hành copy từ FIXED_EXPENSE và FIXED_INCOME
    if (!userCategories || userCategories.length === 0) {
      const defaultCategories = [];

      // Phân tách chi phí
      for (const [level1, level2Array] of Object.entries(FIXED_EXPENSE)) {
        level2Array.forEach(level2 => {
          defaultCategories.push({ chat_id: chatId, transaction_type: 'chi', level_1: level1, level_2: level2 });
        });
      }

      // Phân tách thu nhập
      for (const [level1, level2Array] of Object.entries(FIXED_INCOME)) {
        level2Array.forEach(level2 => {
          defaultCategories.push({ chat_id: chatId, transaction_type: 'thu', level_1: level1, level_2: level2 });
        });
      }

      // Bắn toàn bộ mảng dữ liệu lên Supabase chỉ với 1 lệnh POST
      await supabaseFetch(env, 'categories', 'POST', defaultCategories);
    }

    // CHỐT CHẶN KIỂM TRA LỖI (Giữ nguyên)
    if (!accs || accs.length < 3) {
      return sendMessage(env, chatId, `❌ <b>LỖI NGHIÊM TRỌNG:</b> Không thể ghi dữ liệu vào Supabase.\n\n💡 <b>Vui lòng kiểm tra lại trên Supabase:</b>\n1. Các bảng <code>users</code> và <code>accounts</code> đã được chạy lệnh SQL tạo thành công chưa?\n2. Khóa API và URL cấu hình trong Cloudflare có bị sai ký tự nào không?`);
    }

    const msg = `🐱 <b>DANCIN MMAA - TRỢ LÝ QUẢN LÝ TÀI CHÍNH 6 HŨ BẰNG AI</b>

Chào mừng <b>${name}</b> đến với Dancin MMAA! 🚀 Từ hôm nay, việc quản lý thu chi và tích lũy của bạn sẽ trở nên đơn giản hơn bao giờ hết nhờ sự hỗ trợ của AI.

━━━━━━━━━━━━━━━━━━━

<b>🌟 DANCIN MMAA GIÚP BẠN THẾ NÀO?</b>
1. <b>Ghi chép siêu tốc:</b> Chỉ cần nhắn tin hoặc gửi tin nhắn thoại (Voice) như nói chuyện bình thường. Ví dụ: <i>"ăn sáng 35k"</i>, <i>"tiền lương 15 triệu"</i>, <i>"rút 500k từ ngân hàng sang ví tiền mặt"</i>.
2. <b>Tự động chia 6 Hũ:</b> AI tự động phân tích và phân bổ tiền của bạn vào các hũ tài chính tương ứng (Thiết yếu, Tiết kiệm, Hưởng thụ...) tức thì.

━━━━━━━━━━━━━━━━━━━

<b>🚀 3 BƯỚC ĐỂ BẮT ĐẦU NGAY:</b>
• <b>Bước 1:</b> Thử nhắn tin hoặc gửi một Voice chi tiêu đầu tiên ngay tại đây (Ví dụ: <i>"ăn trưa 50k"</i>).
• <b>Bước 2:</b> Bấm nút <b>Mở App / Dashboard</b> ở góc dưới bên trái màn hình chat để xem ví và các hũ tài chính của bạn.
• <b>Bước 3:</b> Tại ứng dụng, nhấp chọn hũ bất kỳ ➡️ Chọn <b>⚙ Quản lý liên kết</b> để chỉnh sửa danh mục chi tiêu theo thói quen của bạn.

👇 <i>Bắt đầu hành trình tự do tài chính bằng cách ghi chép giao dịch đầu tiên của bạn ngay lúc này nhé!</i>`;
    return sendMessage(env, chatId, msg);
  }

  // 4. LỆNH /BIEUDO
  if (text === '/bieudo') {
    await sendMessage(env, chatId, "📊 Đợi mình một xíu...");
    const chartUrl = `https://quickchart.io/chart?c={type:'doughnut',data:{labels:['Sinh hoạt','Cố định','Đầu tư'],datasets:[{data:[50,30,20]}]}}`;
    await fetch(`https://api.telegram.org/bot${env.TelegramToken}/sendPhoto`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ chat_id: chatId, photo: chartUrl, caption: "📊 Phân bổ dòng tiền tháng này." })
    });
    return;
  }

  // 4b. LỆNH /THEMDANHMUC
  if (text.startsWith('/themdanhmuc')) {
    const parts = text.split(/\s+/);
    if (parts.length < 3) {
      const msg = `⚠️ <b>Sai cú pháp!</b>\n\nVui lòng nhập theo cú pháp:\n<code>/themdanhmuc [chi/thu] [Tên cấp 1], [Tên cấp 2]</code>\n\nVí dụ:\n• <code>/themdanhmuc chi Chi phí sinh hoạt, Trà sữa</code>\n• <code>/themdanhmuc thu Thu nhập, Job phụ 3</code>`;
      return sendMessage(env, chatId, msg);
    }
    
    const typeInput = parts[1].toLowerCase();
    const type = (typeInput === 'thu' || typeInput === 'income') ? 'thu' : 'chi';
    
    const remainingText = parts.slice(2).join(" ");
    const commaIndex = remainingText.indexOf(',');
    if (commaIndex === -1) {
      const msg = `⚠️ <b>Thiếu dấu phẩy phân tách!</b>\n\nVui lòng dùng dấu phẩy (<code>,</code>) để phân tách tên cấp 1 và cấp 2.\n\nVí dụ: <code>/themdanhmuc chi Chi phí sinh hoạt, Trà sữa</code>`;
      return sendMessage(env, chatId, msg);
    }
    
    const rawL1 = remainingText.substring(0, commaIndex).trim();
    const rawL2 = remainingText.substring(commaIndex + 1).trim();
    
    const level1 = mapLevel1(rawL1, type);
    if (!level1) {
      const standardL1s = type === 'chi' 
        ? ["Chi phí sinh hoạt", "Chi phí cố định", "Chi phí phát sinh", "Đầu tư - Tiết kiệm"]
        : ["Thu nhập", "Lãi/Lời", "Thưởng/Quà", "Khác"];
      const msg = `⚠️ <b>Danh mục cấp 1 không hợp lệ!</b>\n\nDanh mục cấp 1 phải thuộc một trong các nhóm:\n${standardL1s.map(l => `• <code>${l}</code>`).join('\n')}`;
      return sendMessage(env, chatId, msg);
    }
    
    if (!rawL2) {
      const msg = `⚠️ <b>Tên danh mục cấp 2 không được để trống!</b>`;
      return sendMessage(env, chatId, msg);
    }
    
    const existing = await supabaseFetch(env, `categories?chat_id=eq.${chatId}&transaction_type=eq.${type}&level_1=eq.${encodeURIComponent(level1)}&level_2=eq.${encodeURIComponent(rawL2)}`);
    if (existing && existing.length > 0) {
      return sendMessage(env, chatId, `💡 Danh mục <b>${level1} ➡️ ${rawL2}</b> đã tồn tại sẵn trong hệ thống.`);
    }
    
    await supabaseFetch(env, 'categories', 'POST', {
      chat_id: chatId,
      transaction_type: type,
      level_1: level1,
      level_2: rawL2
    });
    
    return sendMessage(env, chatId, `✅ <b>Thành công!</b>\n\nĐã thêm danh mục mới vào hệ thống:\n📁 <b>[${type === 'chi' ? 'CHI TÀI KHOẢN' : 'THU NHẬP'}] ${level1} ➡️ ${rawL2}</b>`);
  }

  // 5. GIAO DỊCH QUA AI
  if (message.voice) {
    await sendMessage(env, chatId, "🎙️ Mình đang nghe và dịch ra chữ...");
    const filePath = await getFilePath(env, message.voice.file_id);
    
    if (filePath) {
      const voiceText = await transcribeAudio(env, filePath);
      if (voiceText) {
        await sendMessage(env, chatId, `📝 Bạn vừa nói: "<i>${voiceText}</i>"`);
        // Đẩy phần văn bản vừa dịch được vào lõi AI để xử lý giao dịch
        await processTransaction(chatId, voiceText, env, name);
      } else {
        await sendMessage(env, chatId, "❌ Mình nghe không rõ, bạn có thể nói lại hoặc gõ chữ được không?");
      }
    } else {
      await sendMessage(env, chatId, "❌ Không thể tải file âm thanh từ Telegram.");
    }
  } 
  // Xử lý khi người dùng nhập Text thông thường
  else if (text && !text.startsWith('/')) {
    await processTransaction(chatId, text, env, name);
  }


  // ----------------------------------------------------
  // BẮT SỰ KIỆN: NGƯỜI DÙNG NHẬP DANH MỤC MỚI (FORCE REPLY)
  // ----------------------------------------------------
  if (message.reply_to_message && message.reply_to_message.text?.includes('Mã GD:')) {
    const repliedText = message.reply_to_message.text;
    const newL2 = text.trim(); // Tên danh mục người dùng vừa gõ
    
    // Bóc tách L1 và Transaction ID từ tin nhắn cũ của Bot
    const l1Match = repliedText.match(/nhóm (.*?)\?/);
    const txIdMatch = repliedText.match(/Mã GD: (.*?)\)/);

    if (l1Match && txIdMatch) {
      const l1 = l1Match[1];
      const txId = txIdMatch[1];

      await sendMessage(env, chatId, `⏳ Đang ghi nhớ danh mục mới...`);

      // 1. Tìm xem L1 này thuộc Thu hay Chi (để lưu đúng logic)
      const refCat = await supabaseFetch(env, `categories?chat_id=eq.${chatId}&level_1=ilike.${l1}%25`);
      const txType = (refCat && refCat.length > 0) ? refCat[0].transaction_type : 'chi';

      // 2. Thêm vào bảng categories để AI học cho lần sau
      await supabaseFetch(env, 'categories', 'POST', {
        chat_id: chatId,
        transaction_type: txType,
        level_1: l1,
        level_2: newL2
      });

      // 3. Sửa lại giao dịch gốc
      await supabaseFetch(env, `transactions?id=eq.${txId}`, 'PATCH', { level1: l1, level2: newL2 });

      return sendMessage(env, chatId, `✅ <b>Thành công!</b>\n\nTừ nay mình đã nhớ danh mục: <b>${l1} ➡️ ${newL2}</b>.\nGiao dịch của bạn cũng đã được cập nhật!`);
    }
  }
}

// ==========================================
// LÕI XỬ LÝ GIAO DỊCH VỚI AI COMPONENT (TÍCH HỢP X-QUANG LỖI)
// ==========================================
// Bạn có thể xóa hẳn 2 hằng số FIXED_EXPENSE và FIXED_INCOME ở đầu file đi cho gọn code nhé.

async function processTransaction(chatId, text, env, userName) {
  const accountsData = await supabaseFetch(env, `accounts?chat_id=eq.${chatId}`) || [];
  const accountsList = accountsData.map(a => a.account_name).join(', ');
  
  // Tìm tài khoản mặc định (ví chính) của người dùng
  const defaultAccount = accountsData.find(a => a.is_default === true) || accountsData.find(a => a.account_name === 'Tiền mặt') || accountsData[0];
  const defaultAccountName = defaultAccount ? defaultAccount.account_name : 'Tiền mặt';

  const now = new Date();
  const currentTime = now.toLocaleTimeString('vi-VN', { timeZone: 'Asia/Ho_Chi_Minh', hour: '2-digit', minute: '2-digit' });

  // 🌟 BƯỚC 0: LẤY 10 GIAO DỊCH GẦN ĐÂY ĐỂ AI TỰ HỌC THÓI QUEN CHI TIÊU
  const recentTxs = await supabaseFetch(env, `transactions?chat_id=eq.${chatId}&limit=10&order=created_at.desc`) || [];
  const recentTxsFormatted = recentTxs.map(tx => `- ${tx.date || ''}: ${tx.type === 'chi' ? 'Chi' : 'Thu'} ${parseInt(tx.amount || 0).toLocaleString('vi-VN')}đ vào nhóm [${tx.level1} -> ${tx.level2}] (Ghi chú: ${tx.note || ''})`).join('\n');

  // 🌟 BƯỚC 1: LẤY TỪ ĐIỂN DANH MỤC RIÊNG CỦA USER TỪ SUPABASE
  const userCategories = await supabaseFetch(env, `categories?chat_id=eq.${chatId}`);
  
  let userExpense = {};
  let userIncome = {};

  // 🌟 BƯỚC 2: GOM NHÓM DỮ LIỆU ĐỂ RENDER RA JSON CHO PROMPT
  if (userCategories && userCategories.length > 0) {
    userCategories.forEach(cat => {
      if (cat.transaction_type === 'chi') {
        if (!userExpense[cat.level_1]) userExpense[cat.level_1] = [];
        if (!userExpense[cat.level_1].includes(cat.level_2)) userExpense[cat.level_1].push(cat.level_2);
      } else if (cat.transaction_type === 'thu') {
        if (!userIncome[cat.level_1]) userIncome[cat.level_1] = [];
        if (!userIncome[cat.level_1].includes(cat.level_2)) userIncome[cat.level_1].push(cat.level_2);
      }
    });
  } else {
    // Dự phòng nếu bảng categories bị trống (lỗi bất ngờ), trả về một mảng cơ bản nhất để AI không chết
    userExpense = { "Chi phí sinh hoạt": ["Khác"] };
    userIncome = { "Thu nhập": ["Khác"] };
  }

  // Chuyển đổi thành chuỗi JSON đẹp mắt để nhồi vào Prompt
  const expenseStructure = JSON.stringify(userExpense, null, 2);
  const incomeStructure = JSON.stringify(userIncome, null, 2);

  // 🌟 BƯỚC 3: PROMPT ĐƯỢC CẬP NHẬT TỰ HỌC & SỬA NGỌNG
  const systemPrompt = `Bạn là trợ lý tài chính. Xưng "mình" và "${userName}".
  Thời gian hiện tại của người dùng đang là: ${currentTime}.

  LỊCH SỬ GIAO DỊCH GẦN ĐÂY CỦA NGƯỜI DÙNG (Dùng để tham khảo thói quen chi tiêu khi người dùng viết tắt, nói ngọng hoặc thiếu ý):
  ${recentTxsFormatted || 'Chưa có giao dịch gần đây.'}

  Quy tắc phân loại (BẮT BUỘC tuân thủ chặt chẽ):
  - CHI TIÊU: Dựa vào cấu trúc JSON sau để xác định "level1" và "level2":
  ${expenseStructure}
  - THU NHẬP: Dựa vào cấu trúc JSON sau để xác định "level1" và "level2":
  ${incomeStructure}
  - Nguồn tiền: ${accountsList || 'Tiền mặt'}
  - Tài khoản mặc định (ví chính): Nếu người dùng KHÔNG chỉ định rõ ví thanh toán trong câu (ví dụ chỉ nói 'ăn trưa 50k', 'mua cafe 30k'), bạn BẮT BUỘC phải chọn nguồn tiền là '${defaultAccountName}'. Chỉ khi nào họ nói rõ ví khác (ví dụ: 'thẻ tín dụng', 'tiền mặt', v.v.) thì mới đổi.

  LƯU Ý QUAN TRỌNG VỀ PHÁT ÂM & LỖI CHÍNH TẢ:
  - Giọng nói hoặc tin nhắn của người dùng có thể bị lỗi chính tả hoặc phát âm ngọng (ví dụ Whisper dịch nhầm: 'ăn xáng'/'ăn chửa' -> 'ăn sáng'/'ăn trưa'; 'phê'/'ca phe' -> 'cafe'; 'ngân hằng' -> 'ngân hàng'; 'tiền mẹt'/'tiền mạt' -> 'tiền mặt'). Hãy tự động sửa lỗi và hiểu đúng giao dịch gốc.
  - Hãy dựa vào Lịch sử giao dịch gần đây để phán đoán thói quen chi tiêu. Nếu người dùng nói câu ngắn hoặc ngọng trùng khớp với một thói quen chi tiêu trước đây, hãy ưu tiên phân loại chính xác theo thói quen cũ đó.
  - Chỉ phân loại là "transaction" nếu trong câu nói của người dùng có chứa số tiền cụ thể hoặc có thể suy luận rõ ràng (ví dụ: 'ăn trưa 50k', 'mua cafe ba chục'). Nếu chỉ nói chung chung không có số tiền (ví dụ: 'tôi đi ăn phở', 'tôi mới mua áo khoác'), hãy phân loại là "chat" và phản hồi thân thiện, đồng thời hỏi nhẹ nhàng xem người dùng có muốn ghi chép giao dịch đó kèm số tiền hay không.

  Ý ĐỊNH NGƯỜI DÙNG:
  1. Nếu người dùng muốn GHI CHÉP GIAO DỊCH (ví dụ: 'ăn trưa 50k', 'lương 10tr', 'chuyển 500k từ ngân hàng sang tiền mặt'):
     Trả về JSON Schema:
     {
       "action": "transaction",
       "type": "thu" | "chi" | "chuyen_tien",
       "amount": number,
       "level1": "Tên cấp 1 (lấy chính xác từ sơ đồ)",
       "level2": "Tên cấp 2 (lấy chính xác từ sơ đồ)",
       "account_name": "nguồn tiền",
       "transfer_to_account": "ví nhận (nếu type=chuyen_tien)",
       "note": "ghi chú",
       "bot_message": "câu nhận xét ngắn, dễ thương"
     }
  2. Nếu người dùng muốn THÊM MỚI DANH MỤC CẤP 2 (ví dụ: 'thêm danh mục Trà sữa vào Chi phí sinh hoạt', 'Tôi muốn thêm danh mục Làm đẹp cho Chi phí sinh hoạt', 'thêm nhóm Ăn vặt cho chi phát sinh'):
     Trả về JSON Schema:
     {
       "action": "add_category",
       "type": "thu" | "chi",
       "level1": "Tên danh mục cấp 1 tương ứng (phải chọn từ sơ đồ danh mục ở trên, ví dụ 'Chi phí sinh hoạt' hoặc 'Chi phí phát sinh')",
       "level2": "Tên danh mục cấp 2 mới cần thêm",
       "bot_message": "câu xác nhận ngắn dễ thương"
     }
  3. Nếu người dùng muốn XÓA DANH MỤC CẤP 2 (ví dụ: 'xóa danh mục Làm đẹp khỏi Chi phí sinh hoạt', 'xóa nhóm Trà sữa trong Chi phí sinh hoạt', 'bỏ danh mục này đi'):
     Trả về JSON Schema:
     {
       "action": "delete_category",
       "type": "thu" | "chi",
       "level1": "Tên danh mục cấp 1 tương ứng (phải chọn từ sơ đồ danh mục ở trên)",
       "level2": "Tên danh mục cấp 2 cần xóa",
       "bot_message": "câu xác nhận ngắn dễ thương"
     }
  4. Nếu người dùng muốn YÊU CẦU TÍNH NĂNG MỚI hoặc GÓP Ý PHÁT TRIỂN thêm chức năng chưa có (ví dụ: 'tôi muốn xuất file excel báo cáo', 'mong app có tính năng chia sẻ ví với người yêu', 'app có tính năng nhắc nhở chi tiêu không?'):
     Trả về JSON Schema:
     {
       "action": "feature_request",
       "feature_description": "Nội dung yêu cầu tính năng chi tiết từ người dùng (tự tóm tắt ngắn gọn và rõ nghĩa bằng tiếng Việt)",
       "bot_message": "câu phản hồi dễ thương báo là đã ghi nhận ý kiến và sẽ chuyển tới Admin ngay để xem xét phát triển"
     }
  5. Nếu người dùng chỉ chào hỏi, nói chuyện phiếm, chia sẻ cảm xúc, hỏi đáp thông thường (ví dụ: 'chào bạn', 'hôm nay mệt quá', 'tại sao nên tiết kiệm tiền?'):
     Trả về JSON Schema:
     {
       "action": "chat",
       "bot_message": "câu trả lời tự nhiên, thân thiện, đồng cảm hoặc tư vấn tài chính ngắn gọn phù hợp"
     }

  Chỉ trả về JSON, không Markdown.`;

  try {
    const groqKey = env.GROQ_API_KEY || env.GroqAPIkey || env.groq_api_key;
    if (!groqKey) {
      return sendMessage(env, chatId, "❌ <b>Lỗi cấu hình:</b> Chưa tìm thấy biến môi trường API Key của Groq trong Cloudflare.");
    }

    const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${groqKey}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'llama-3.3-70b-versatile',
        messages: [{ role: 'system', content: systemPrompt }, { role: 'user', content: text }],
        response_format: { type: "json_object" }
      })
    });

    if (!response.ok) {
      const errText = await response.text();
      return sendMessage(env, chatId, `❌ <b>Lỗi kết nối AI (Groq):</b>\n<code>${errText}</code>`);
    }

    const aiResult = await response.json();
    let rawContent = aiResult.choices[0].message.content;
    rawContent = rawContent.replace(/```json/g, '').replace(/```/g, '').trim();
    const data = JSON.parse(rawContent);

    if (data.action === 'transaction') {
      const insertData = {
        chat_id: chatId,
        type: data.type,
        amount: data.amount,
        date: new Date().toLocaleDateString('en-GB', { timeZone: 'Asia/Ho_Chi_Minh' }),
        level1: data.level1,
        level2: data.level2,
        account_name: data.account_name,
        transfer_to_account: data.transfer_to_account || null,
        note: data.note
      };

      const insertResponse = await supabaseFetch(env, 'transactions', 'POST', insertData);
      const realTxId = (insertResponse && insertResponse.length > 0) ? insertResponse[0].id : Date.now();
      const typeIcon = data.type === 'chi' ? '🔴' : data.type === 'thu' ? '🟢' : '🔄';
      
      const reply = `${typeIcon} <b>Mình đã ghi lại giúp bạn!</b>\n💰 Số tiền: <b>${data.amount.toLocaleString('vi-VN')} đ</b>\n📂 Danh mục: ${data.level1} ➡️ ${data.level2}\n💳 Từ ví: ${data.account_name} ${data.type === 'chuyen_tien' ? '➡️ ' + data.transfer_to_account : ''}\n📝 Ghi chú: ${data.note}\n\n💬 <i>"${data.bot_message}"</i>`;
      await sendMessage(env, chatId, reply, { inline_keyboard: [[{ text: '✏️ Sửa lại xíu', callback_data: `edit_${realTxId}` }]] });
    }
    else if (data.action === 'add_category') {
      const type = data.type;
      const level1 = data.level1;
      const level2 = data.level2;

      // Kiểm tra trùng
      const existing = await supabaseFetch(env, `categories?chat_id=eq.${chatId}&transaction_type=eq.${type}&level_1=eq.${encodeURIComponent(level1)}&level_2=eq.${encodeURIComponent(level2)}`);
      if (existing && existing.length > 0) {
        return sendMessage(env, chatId, `💡 Danh mục <b>${level1} ➡️ ${level2}</b> đã tồn tại sẵn trong hệ thống.`);
      }

      await supabaseFetch(env, 'categories', 'POST', {
        chat_id: chatId,
        transaction_type: type,
        level_1: level1,
        level_2: level2
      });

      return sendMessage(env, chatId, `✅ <b>Đã thêm danh mục thành công bằng AI!</b>\n\n📁 <b>[${type === 'chi' ? 'CHI TÀI KHOẢN' : 'THU NHẬP'}] ${level1} ➡️ ${level2}</b>\n\n💬 <i>"${data.bot_message}"</i>`);
    }
    else if (data.action === 'delete_category') {
      const type = data.type;
      const level1 = data.level1;
      const level2 = data.level2;

      // Gọi DELETE xuống Supabase
      await supabaseFetch(env, `categories?chat_id=eq.${chatId}&transaction_type=eq.${type}&level_1=eq.${encodeURIComponent(level1)}&level_2=eq.${encodeURIComponent(level2)}`, 'DELETE');

      return sendMessage(env, chatId, `✅ <b>Đã xóa danh mục thành công bằng AI!</b>\n\n🗑️ <b>[${type === 'chi' ? 'CHI TÀI KHOẢN' : 'THU NHẬP'}] ${level1} ➡️ ${level2}</b>\n\n💬 <i>"${data.bot_message}"</i>`);
    }
    else if (data.action === 'feature_request') {
      const featureDesc = data.feature_description;

      // 1. Lưu vào bảng feature_requests của Supabase
      await supabaseFetch(env, 'feature_requests', 'POST', {
        chat_id: chatId,
        user_name: userName,
        feature_description: featureDesc
      });

      // 2. Phản hồi người dùng
      await sendMessage(env, chatId, `📝 <b>Đã ghi nhận ý kiến góp ý của bạn!</b>\n\n💡 Yêu cầu: <i>"${featureDesc}"</i>\n\n💬 <i>"${data.bot_message}"</i>`);

      // 3. Báo cho Admin (nếu có ADMIN_CHAT_ID)
      const adminChatId = env.ADMIN_CHAT_ID || env.admin_chat_id;
      if (adminChatId) {
        const adminMsg = `🔔 <b>YÊU CẦU TÍNH NĂNG MỚI!</b>\n\n👤 Người dùng: <b>${userName}</b> (ID: <code>${chatId}</code>)\n💡 Nội dung: <i>${featureDesc}</i>\n📅 Thời gian: ${new Date().toLocaleString('vi-VN', { timeZone: 'Asia/Ho_Chi_Minh' })}`;
        await sendMessage(env, adminChatId, adminMsg);
      }
    }
    else if (data.action === 'chat') {
      return sendMessage(env, chatId, data.bot_message);
    }

  } catch (error) {
    await sendMessage(env, chatId, `🛠 <b>Phát hiện lỗi xử lý:</b>\n<code>${error.message}</code>\n\nBạn hãy gửi mã lỗi này cho tôi nhé!`);
  }
}

// ==========================================
// XỬ LÝ LỆNH CALLBACK QUERY (Nút bấm Inline)
// ==========================================
async function handleCallbackQuery(callbackQuery, env, ctx) {
  const data = callbackQuery.data;
  const chatId = callbackQuery.message.chat.id.toString();
  const messageId = callbackQuery.message.message_id;

  // Hợp nhất cấu trúc danh mục từ hằng số đầu file
  const allCategories = { ...FIXED_EXPENSE, ...FIXED_INCOME };

  // 1. CHỌN DANH MỤC CẤP 1 (Sử dụng INDEX để tối ưu dung lượng nút bấm dưới 64 bytes)
  if (data.startsWith('edit_')) {
    const txId = data.split('_')[1];

    const uniqueL1 = Object.keys(allCategories);
    const keyboard = uniqueL1.map((l1, idx) => ([{ 
      text: `📂 ${l1}`, 
      callback_data: `e2|${txId}|${idx}` 
    }]));

    await editMessageText(env, chatId, messageId, `✏️ <b>Đang sửa giao dịch...</b>\n\nHãy chọn <b>Danh mục Cấp 1</b> mới:`, { inline_keyboard: keyboard });
    return;
  }

  // NÚT QUAY LẠI DANH MỤC CẤP 1 (Khi người dùng bấm nút Quay lại ở tầng Cấp 2)
  if (data.startsWith('back_l1|')) {
    const txId = data.split('|')[1];
    
    const uniqueL1 = Object.keys(allCategories);
    const keyboard = uniqueL1.map((l1, idx) => ([{ 
      text: `📂 ${l1}`, 
      callback_data: `e2|${txId}|${idx}` 
    }]));

    await editMessageText(env, chatId, messageId, `👇 Hãy chọn <b>Danh mục Cấp 1</b>:`, { inline_keyboard: keyboard });
    return;
  }

  // 2. HIỂN THỊ DANH MỤC CẤP 2 (Sau khi đã chọn Cấp 1)
  if (data.startsWith('e2|')) {
    const parts = data.split('|');
    const txId = parts[1];
    const l1Idx = parseInt(parts[2]);

    const uniqueL1 = Object.keys(allCategories);
    const fullL1Key = uniqueL1[l1Idx];
    if (!fullL1Key) return;

    const uniqueL2 = allCategories[fullL1Key] || [];
    const keyboard = [];
    let row = [];
    
    uniqueL2.forEach((l2, l2Idx) => {
      // Truyền thông tin es | id giao dịch | vị trí cấp 1 | vị trí cấp 2
      row.push({ 
        text: l2, 
        callback_data: `es|${txId}|${l1Idx}|${l2Idx}` 
      });
      
      if (row.length === 2 || l2Idx === uniqueL2.length - 1) {
        keyboard.push(row);
        row = [];
      }
    });

    // Tích hợp thanh điều hướng: Quay lại & Thêm mới/Khác xuống cuối menu
    keyboard.push([
      { text: '🔙 Quay lại', callback_data: `back_l1|${txId}` },
      { text: '➕ Thêm mới/Khác', callback_data: `add_l2|${txId}` }
    ]);

    await editMessageText(env, chatId, messageId, `📂 Đã chọn: <b>${fullL1Key}</b>\n\n👇 Hãy chọn tiếp <b>Danh mục Cấp 2</b>:`, { inline_keyboard: keyboard });
    return;
  }

  // 3. HÀNH ĐỘNG TIẾP THEO: XÁC NHẬN LƯU KHI CHỌN DANH MỤC CẤP 2
  if (data.startsWith('es|')) {
    const parts = data.split('|');
    const txId = parts[1];
    const l1Idx = parseInt(parts[2]);
    const l2Idx = parseInt(parts[3]);

    // Dựa vào Index để giải mã ngược lại tên danh mục đầy đủ
    const uniqueL1 = Object.keys(allCategories);
    const fullL1 = uniqueL1[l1Idx];
    const fullL2 = allCategories[fullL1]?.[l2Idx];

    if (!fullL1 || !fullL2) return;

    // Gửi lệnh PATCH để cập nhật trực tiếp danh mục mới vào Supabase
    await supabaseFetch(env, `transactions?id=eq.${txId}`, 'PATCH', {
      level1: fullL1,
      level2: fullL2
    });

    // Cập nhật lại tin nhắn thông báo thành công rực rỡ
    await editMessageText(env, chatId, messageId, `✅ <b>Cập nhật danh mục thành công!</b>\n\n📂 Danh mục mới: <b>${fullL1} ➡️ ${fullL2}</b>\n\n<i>Dancin MMAA đã đồng bộ dữ liệu chuẩn xác vào hệ thống!</i>`);
    return;
  }

  // 4. XỬ LÝ KHI BẤM NÚT "THÊM MỚI/KHÁC"
  if (data.startsWith('add_l2|')) {
    const txId = data.split('|')[1];

    // Lúc này mới chính thức XÓA giao dịch cũ trên Supabase vì người dùng muốn hủy hẳn để làm lại
    await supabaseFetch(env, `transactions?id=eq.${txId}`, 'DELETE');

    await editMessageText(env, chatId, messageId, `🗑️ <b>Đã hủy giao dịch cũ phân loại sai!</b>\n\n✍️ Giờ bạn hãy nhập chữ hoặc gửi voice lại một câu giao dịch mới (Ví dụ: "Ăn phở gà 50k ví tiền mặt") để mình xử lý lại từ đầu nhé.`);
    return;
  }
}
